public class AulaParte2 {

        reu.setPeso(scanf.nextDouble());

        reu.setAltura(scanf.nextDouble());

        advogado.setNome(scanf.next());
        System.out.println("Digite a idade do advogado: ");
        advogado.setIdade(scanf.nextInt());
        System.out.println("Digite o peso do advogado: ");
        advogado.setPeso(scanf.nextDouble());
        System.out.println("Digite a altura do advogado: ");
        advogado.setAltura(scanf.nextDouble());

        promotor.setNome(scanf.next());

        promotor.setIdade(scanf.nextInt());
        System.out.println("Digite o peso do promotor: ");
        promotor.setPeso(scanf.nextDouble());
        System.out.println("Digite a altura do promotor: ");
        promotor.setAltura(scanf.nextDouble());

        System.out.print(reu.getNome());

        System.out.print(reu.getIdade());
        
        System.out.println("");
        
        System.out.print("O advogado se chama ");
        System.out.print(advogado.getNome());
