package com.mycompany.aulaparte3;

import java.util.Random;

/**
 *
 * @author Vitor
 */
public class AulaParte3 {
    static int vetor[] = { 1, 2, 3, 4, 5, 6 };
    
    public static void vetorIteracao(int vet[]) {
        Random random = new Random();
        for(int i = 0; i < vet.length; i++) {
            System.out.println("Vet anterior: " + vet[i]);
            vet[i] = random.nextInt(10);
            System.out.println("Vet novo: " + vet[i]);
        }
    }

    public static void main(String[] args) {
        for(int i = 0; i < vetor.length; i++) {
            System.out.println("Vetor anterior: " + vetor[i]);
        }
        
        vetorIteracao(vetor);
        
        for(int i = 0; i < vetor.length; i++) {
            System.out.println("Vetor novo: " + vetor[i]);
        }
    }
}
